List 100 phone numbers in separated parts: country code, area code and line number

SELECT LEFT(content,1) AS countryCode, SUBSTRING(content, 3, 3) AS areaCode, RIGHT(content, 8) AS lineNumber FROM Phone WHERE content LIKE "1-%" LIMIT 100;
