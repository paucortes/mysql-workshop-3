List all Account emails with AddressBook containing Phone numbers with a country code

SELECT email FROM Account JOIN AddressBook ON Account.id = AddressBook.accountId JOIN Entry ON AddressBook.id = Entry.addressBookId JOIN Phone ON Entry.id = Phone.entryId AND Phone.content LIKE "1-%" GROUP BY AddressBook.id;
